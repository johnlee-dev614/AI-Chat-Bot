# simple memory
chat_history = []

# 💬 Chat endpoint
@app.api_route("/chat", methods=["GET", "POST"])
def chat(message: str = Query(None), body: dict = Body(None)):

    # Determine message source (GET query or POST body)
    user_message = message or (body.get("message") if body else None)
    if not user_message:
        return {"error": "No message provided"}

    chat_history.append({"role": "user", "content": user_message})
    messages = [{"role": "system", "content": SYSTEM_PROMPT}] + chat_history[-10:]

    # Call OpenRouter API
    ai_response = requests.post(
        "https://openrouter.ai/api/v1/chat/completions",
        headers={"Authorization": f"Bearer {OPENROUTER_API_KEY}"},
        json={
            "model": "meta-llama/llama-3-8b-instruct",
            "messages": messages,
            "temperature": 0.95,
            "max_tokens": 80
        }
    )

    ai_data = ai_response.json()
    if "choices" not in ai_data:
        return {"error": ai_data}

    ai_reply = ai_data["choices"][0]["message"]["content"]
    chat_history.append({"role": "assistant", "content": ai_reply})

    # ElevenLabs voice
    voice_response = requests.post(
        f"https://api.elevenlabs.io/v1/text-to-speech/{VOICE_ID}",
        headers={
            "xi-api-key": ELEVENLABS_API_KEY,
            "Content-Type": "application/json"
        },
        json={
            "text": ai_reply,
            "model_id":"eleven_monolingual_v1",
            "voice_settings":{"stability":0.3,"similarity_boost":0.85}
        }
    )

    # Encode audio as Base64
    audio_base64 = base64.b64encode(voice_response.content).decode("utf-8")

    return {"text": ai_reply, "audio": audio_base64}


# 🔊 Voice-only endpoint
@app.api_route("/voice", methods=["GET", "POST"])
def voice(text: str = Query(None), body: dict = Body(None)):

    text_input = text or (body.get("text") if body else None)
    if not text_input:
        return {"error": "No text provided"}

    eleven_response = requests.post(
        f"https://api.elevenlabs.io/v1/text-to-speech/{VOICE_ID}",
        headers={
            "xi-api-key": ELEVENLABS_API_KEY,
            "Content-Type": "application/json"
        },
        json={
            "text": text_input,
            "model_id":"eleven_monolingual_v1",
            "voice_settings":{"stability":0.3,"similarity_boost":0.85}
        }
    )

    audio_base64 = base64.b64encode(eleven_response.content).decode("utf-8")
    return {"audio": audio_base64}
